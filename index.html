Quick Example

// Assign role to user
$user->assignRole('admin');

if ($user->hasPermission('posts.edit')) {

Installation

Requirements

  • PHP 8.2 or higher
  • Laravel 10.x, 11.x, or 12.x
  • Composer

Step 1: Install via Composer

composer require rbac-suite/omni-access

Step 2: Run Installation Command

# Install with default data
php artisan omni-access:install --seed

# Install without seeding
php artisan omni-access:install

Step 3: Add Traits to User Model

<?php

namespace App\Models;

use Illuminate\Foundation\Auth\User as Authenticatable;
use RbacSuite\OmniAccess\Traits\HasRoles;
use RbacSuite\OmniAccess\Traits\HasPermissions;

class User extends Authenticatable
{
    use HasRoles, HasPermissions;
    
    // Your existing code...
}

Working with Roles

Create Roles

use RbacSuite\OmniAccess\Models\Role;

// Create a new role
$role = Role::create([
    'name' => 'Editor',
    'slug' => 'editor',
    'description' => 'Can edit content',
    'is_default' => false
]);

// Using the manager
$role = app('omni-access')->createRole([
    'name' => 'Moderator',
    'slug' => 'moderator'
]);

Assign Roles to Users

// Single role
$user->assignRole('admin');

// Multiple roles
$user->assignRole('editor', 'moderator');

// Using role instance
$adminRole = Role::where('slug', 'admin')->first();
$user->assignRole($adminRole);

// Remove role
$user->removeRole('editor');

// Sync roles (replace all existing)
$user->syncRoles(['admin', 'editor']);

Check User Roles

// Check single role
if ($user->hasRole('admin')) {
    // User is admin
}

// Check any role
if ($user->hasAnyRole(['admin', 'editor'])) {
    // User has at least one role
}

// Check all roles
if ($user->hasAllRoles(['admin', 'editor'])) {
    // User has both roles
}

// Check super admin
if ($user->isSuperAdmin()) {
    // User is super admin
}

// Get all user roles
$roles = $user->roles; // Collection of Role models

Role Permissions

// Assign permissions to role
$role->givePermissionTo('posts.edit', 'posts.delete');

// Remove permission from role
$role->revokePermissionTo('posts.delete');

// Sync permissions
$role->syncPermissions(['posts.view', 'posts.edit']);

// Check if role has permission
if ($role->hasPermission('posts.edit')) {
    // Role has permission
}

Working with Permissions

Create Permissions

use RbacSuite\OmniAccess\Models\Permission;
use RbacSuite\OmniAccess\Models\Group;

// Create permission group
$group = Group::create([
    'name' => 'Post Management',
    'slug' => 'post-management',
    'description' => 'Permissions for managing posts',
    'sort_order' => 1
]);

// Create permission
$permission = Permission::create([
    'name' => 'Edit Posts',
    'slug' => 'posts.edit',
    'description' => 'Can edit posts',
    'group_id' => $group->id
]);

Direct User Permissions

// Give direct permission to user
$user->givePermissionTo('special.access');

// Revoke direct permission
$user->revokePermissionTo('special.access');

// Check permission (includes role permissions)
if ($user->hasPermission('posts.edit')) {
    // User can edit posts
}

// Check multiple permissions
if ($user->hasAnyPermission(['posts.edit', 'posts.delete'])) {
    // Has at least one permission
}

if ($user->hasAllPermissions(['posts.edit', 'posts.publish'])) {
    // Has all permissions
}

Permission Groups

// Get all groups with permissions
$groups = Group::with('permissions')->orderBy('sort_order')->get();

// Get permissions by group
$group = Group::where('slug', 'post-management')->first();
$permissions = $group->permissions; // All permissions in this group

// Get user permissions grouped
$grouped = $user->getGroupedPermissions();
// Returns: ['Post Management' => [...permissions], 'User Management' => [...]]

Permission Helpers

// Get all user permissions (direct + from roles)
$permissions = $user->getAllPermissions();

// Check model-based permission
if ($user->canOnModel('edit', Post::class)) {
    // User can edit posts
}

// Cache operations
$user->refreshPermissionsCache(); // Refresh cached permissions

Middleware

Available Middleware

  • role - Check if user has specific role(s)
  • permission - Check if user has specific permission(s)
  • role_or_permission - Check if user has role OR permission

Role Middleware

// Single role
Route::middleware(['auth', 'role:admin'])->group(function () {
    Route::get('/admin', [AdminController::class, 'index']);
});

// Multiple roles (user must have at least one)
Route::middleware('role:admin,editor')->group(function () {
    Route::resource('posts', PostController::class);
});

// In controller
public function __construct()
{
    $this->middleware('role:admin')->except(['index', 'show']);
}

Permission Middleware

// Single permission
Route::middleware(['auth', 'permission:posts.edit'])->group(function () {
    Route::get('/posts/{post}/edit', [PostController::class, 'edit']);
});

// Multiple permissions
Route::middleware('permission:posts.edit,posts.delete')->group(function () {
    Route::put('/posts/{post}', [PostController::class, 'update']);
    Route::delete('/posts/{post}', [PostController::class, 'destroy']);
});

Role OR Permission Middleware

// User needs either admin role OR posts.manage permission
Route::middleware('role_or_permission:admin,posts.manage')->group(function () {
    Route::resource('posts', PostController::class);
});

// Multiple checks
Route::middleware('role_or_permission:admin,editor,posts.publish')
    ->post('/posts/{post}/publish', [PostController::class, 'publish']);

Controller: app/Http/Controllers/Admin/RoleController.php

<?php

namespace App\Http\Controllers\Admin;

use App\Http\Controllers\Controller;
use Illuminate\Http\Request;
use RbacSuite\OmniAccess\Models\Role;
use RbacSuite\OmniAccess\Models\Permission;
use RbacSuite\OmniAccess\Models\Group;
use RbacSuite\OmniAccess\Services\CacheService;

class RoleController extends Controller
{
    protected $cacheService;

    public function __construct(CacheService $cacheService)
    {
        $this->cacheService = $cacheService;
        
        // Apply middleware
        $this->middleware('auth');
        $this->middleware('permission:roles.view')->only(['index', 'show']);
        $this->middleware('permission:roles.create')->only(['create', 'store']);
        $this->middleware('permission:roles.edit')->only(['edit', 'update']);
        $this->middleware('permission:roles.delete')->only('destroy');
    }

    /**
     * Display a listing of roles
     */
    public function index(Request $request)
    {
        $query = Role::withCount(['users', 'permissions']);

        // Search functionality
        if ($request->filled('search')) {
            $search = $request->get('search');
            $query->where(function ($q) use ($search) {
                $q->where('name', 'like', "%{$search}%")
                  ->orWhere('slug', 'like', "%{$search}%")
                  ->orWhere('description', 'like', "%{$search}%");
            });
        }

        // Sort functionality
        $sortField = $request->get('sort', 'name');
        $sortDirection = $request->get('direction', 'asc');
        
        if (in_array($sortField, ['name', 'slug', 'created_at', 'users_count', 'permissions_count'])) {
            $query->orderBy($sortField, $sortDirection);
        }

        $roles = $query->paginate(10)->withQueryString();

        return view('admin.roles.index', compact('roles'));
    }

    /**
     * Show the form for creating a new role
     */
    public function create()
    {
        $permissions = Permission::with('group')
            ->get()
            ->groupBy('group.name');
        
        return view('admin.roles.create', compact('permissions'));
    }

    /**
     * Store a newly created role
     */
    public function store(Request $request)
    {
        $validated = $request->validate([
            'name' => 'required|string|max:50',
            'slug' => 'required|string|max:50|unique:roles,slug|regex:/^[a-z0-9\-]+$/',
            'description' => 'nullable|string|max:255',
            'is_default' => 'boolean',
            'permissions' => 'nullable|array',
            'permissions.*' => 'exists:permissions,id'
        ]);

        // Create role
        $role = Role::create([
            'name' => $validated['name'],
            'slug' => $validated['slug'],
            'description' => $validated['description'],
            'is_default' => $request->boolean('is_default')
        ]);

        // Assign permissions
        if (!empty($validated['permissions'])) {
            $role->permissions()->sync($validated['permissions']);
        }

        // Clear cache
        $this->cacheService->forgetRole($role->id);

        return redirect()
            ->route('admin.roles.index')
            ->with('success', 'Role created successfully!');
    }

    /**
     * Display the specified role
     */
    public function show(Role $role)
    {
        $role->load(['permissions.group', 'users']);
        
        $groupedPermissions = $role->permissions
            ->groupBy('group.name');

        $recentUsers = $role->users()
            ->latest('role_user.created_at')
            ->limit(5)
            ->get();

        return view('admin.roles.show', compact('role', 'groupedPermissions', 'recentUsers'));
    }

    /**
     * Show the form for editing the specified role
     */
    public function edit(Role $role)
    {
        // Prevent editing super-admin role
        if ($role->slug === config('omni-access.super_admin_role')) {
            return redirect()
                ->route('admin.roles.index')
                ->with('error', 'Cannot edit super admin role!');
        }

        $permissions = Permission::with('group')
            ->get()
            ->groupBy('group.name');
        
        $rolePermissions = $role->permissions->pluck('id')->toArray();

        return view('admin.roles.edit', compact('role', 'permissions', 'rolePermissions'));
    }

    /**
     * Update the specified role
     */
    public function update(Request $request, Role $role)
    {
        // Prevent editing super-admin role
        if ($role->slug === config('omni-access.super_admin_role')) {
            return redirect()
                ->route('admin.roles.index')
                ->with('error', 'Cannot edit super admin role!');
        }

        $validated = $request->validate([
            'name' => 'required|string|max:50',
            'slug' => 'required|string|max:50|unique:roles,slug,' . $role->id . '|regex:/^[a-z0-9\-]+$/',
            'description' => 'nullable|string|max:255',
            'is_default' => 'boolean',
            'permissions' => 'nullable|array',
            'permissions.*' => 'exists:permissions,id'
        ]);

        // Update role
        $role->update([
            'name' => $validated['name'],
            'slug' => $validated['slug'],
            'description' => $validated['description'],
            'is_default' => $request->boolean('is_default')
        ]);

        // Sync permissions
        $role->permissions()->sync($validated['permissions'] ?? []);

        // Clear cache
        $this->cacheService->forgetRole($role->id);

        return redirect()
            ->route('admin.roles.show', $role)
            ->with('success', 'Role updated successfully!');
    }

    /**
     * Remove the specified role
     */
    public function destroy(Role $role)
    {
        // Prevent deleting super-admin role
        if ($role->slug === config('omni-access.super_admin_role')) {
            return response()->json([
                'success' => false,
                'message' => 'Cannot delete super admin role!'
            ], 403);
        }

        // Prevent deleting default role
        if ($role->is_default) {
            return response()->json([
                'success' => false,
                'message' => 'Cannot delete default role!'
            ], 403);
        }

        // Check if role has users
        if ($role->users()->count() > 0) {
            return response()->json([
                'success' => false,
                'message' => 'Cannot delete role with assigned users!'
            ], 403);
        }

        $role->delete();
        
        // Clear cache
        $this->cacheService->forgetRole($role->id);

        if (request()->wantsJson()) {
            return response()->json([
                'success' => true,
                'message' => 'Role deleted successfully!'
            ]);
        }

        return redirect()
            ->route('admin.roles.index')
            ->with('success', 'Role deleted successfully!');
    }

    /**
     * Clone a role
     */
    public function clone(Role $role)
    {
        $newRole = $role->replicate();
        $newRole->name = $role->name . ' (Copy)';
        $newRole->slug = $role->slug . '-copy-' . time();
        $newRole->is_default = false;
        $newRole->save();

        // Clone permissions
        $newRole->permissions()->sync($role->permissions);

        return redirect()
            ->route('admin.roles.edit', $newRole)
            ->with('success', 'Role cloned successfully!');
    }
}

    /**
     * Display a listing of permissions
     */
    public function index(Request $request)
    {
        $query = Permission::with('group')->withCount('roles');

        // Filter by group
        if ($request->filled('group')) {
            $query->where('group_id', $request->get('group'));
        }

        // Search functionality
        if ($request->filled('search')) {
            $search = $request->get('search');
            $query->where(function ($q) use ($search) {
                $q->where('name', 'like', "%{$search}%")
                  ->orWhere('slug', 'like', "%{$search}%")
                  ->orWhere('description', 'like', "%{$search}%");
            });
        }

        // Sort functionality
        $sortField = $request->get('sort', 'name');
        $sortDirection = $request->get('direction', 'asc');
        
        if (in_array($sortField, ['name', 'slug', 'created_at', 'roles_count'])) {
            $query->orderBy($sortField, $sortDirection);
        } else {
            $query->leftJoin('permission_groups', 'permissions.group_id', '=', 'permission_groups.id')
                  ->orderBy('permission_groups.sort_order')
                  ->orderBy('permissions.name');
        }

        $permissions = $query->paginate(15)->withQueryString();
        $groups = Group::orderBy('sort_order')->get();

        return view('admin.permissions.index', compact('permissions', 'groups'));
    }

        $groups = Group::orderBy('sort_order')->get();

    }

    /**
     * Store a newly created permission
     */
    public function store(Request $request)
    {
        $validated = $request->validate([
            'name' => 'required|string|max:50',
            'slug' => 'required|string|max:50|unique:permissions,slug|regex:/^[a-z0-9\.\-\_]+$/',
            'group_id' => 'nullable|exists:permission_groups,id',
            'description' => 'nullable|string|max:255',
            'roles' => 'nullable|array',
            'roles.*' => 'exists:roles,id'
        ]);

        // Create permission
        $permission = Permission::create([
            'name' => $validated['name'],
            'slug' => $validated['slug'],
            'group_id' => $validated['group_id'],
            'description' => $validated['description']
        ]);

        // Assign to roles
        if (!empty($validated['roles'])) {
            $permission->roles()->sync($validated['roles']);
        }

        // Clear cache
        $this->cacheService->forgetPermission($permission->id);

        if ($request->input('create_another')) {
            return redirect()
                ->route('admin.permissions.create')
                ->with('success', 'Permission created successfully! You can add another.');
        }

        return redirect()
            ->route('admin.permissions.index')
            ->with('success', 'Permission created successfully!');
    }

    /**
     * Update the specified permission
     */
    public function update(Request $request, Permission $permission)
    {
        $validated = $request->validate([
            'name' => 'required|string|max:50',
            'slug' => 'required|string|max:50|unique:permissions,slug,' . $permission->id . '|regex:/^[a-z0-9\.\-\_]+$/',
            'group_id' => 'nullable|exists:permission_groups,id',
            'description' => 'nullable|string|max:255',
            'roles' => 'nullable|array',
            'roles.*' => 'exists:roles,id'
        ]);

        // Update permission
        $permission->update([
            'name' => $validated['name'],
            'slug' => $validated['slug'],
            'group_id' => $validated['group_id'],
            'description' => $validated['description']
        ]);

        // Sync roles
        $permission->roles()->sync($validated['roles'] ?? []);

        // Clear cache
        $this->cacheService->forgetPermission($permission->id);

        if ($request->input('save_and_continue')) {
            return redirect()
                ->route('admin.permissions.edit', $permission)
                ->with('success', 'Permission updated successfully!');
        }

        return redirect()
            ->route('admin.permissions.show', $permission)
            ->with('success', 'Permission updated successfully!');
    }

    /**
     * Remove the specified permission
     */
    public function destroy(Permission $permission)
    {
        // Check if permission has roles
        if ($permission->roles()->count() > 0) {
            return response()->json([
                'success' => false,
                'message' => 'Cannot delete permission that is assigned to roles!'
            ], 403);
        }

        $permission->delete();
        
        // Clear cache
        $this->cacheService->forgetPermission($permission->id);

        if (request()->wantsJson()) {
            return response()->json([
                'success' => true,
                'message' => 'Permission deleted successfully!'
            ]);
        }

        return redirect()
            ->route('admin.permissions.index')
            ->with('success', 'Permission deleted successfully!');
    }

// Permission-specific JavaScript functions

// Select all roles
function selectAllRoles() {
    document.querySelectorAll('.role-checkbox').forEach(cb => cb.checked = true);
    updateRoleCount?.();
}

// Clear all roles
function selectNoRoles() {
    document.querySelectorAll('.role-checkbox').forEach(cb => cb.checked = false);
    updateRoleCount?.();
}

// Update role count
function updateRoleCount() {
    const count = document.querySelectorAll('.role-checkbox:checked').length;
    const counter = document.getElementById('selectedCount');
    if (counter) {
        counter.textContent = count;
    }
}

// Fill permission form with common patterns
function fillPermission(name, slug) {
    const nameInput = document.getElementById('name');
    const slugInput = document.getElementById('slug');
    
    if (nameInput && nameInput.value === '') {
        nameInput.value = name + ' ';
        nameInput.focus();
        nameInput.setSelectionRange(nameInput.value.length, nameInput.value.length);
    }
    
    if (slugInput && slugInput.value === '') {
        slugInput.value = slug;
    }
}

// Delete single permission
function deletePermission(id, name) {
    if (!confirm(`Are you sure you want to delete the permission "${name}"?`)) {
        return;
    }

    fetch(`/admin/permissions/${id}`, {
        method: 'DELETE',
        headers: {
            'X-CSRF-TOKEN': document.querySelector('meta[name="csrf-token"]').content,
            'Accept': 'application/json',
            'Content-Type': 'application/json'
        }
    })
    .then(response => response.json())
    .then(data => {
        if (data.success) {
            location.reload();
        } else {
            alert(data.message || 'Error deleting permission');
        }
    })
    .catch(error => {
        console.error('Error:', error);
        alert('An error occurred while deleting the permission');
    });
}

// Clone permission
function clonePermission(id) {
    if (confirm('Clone this permission?')) {
        const form = document.createElement('form');
        form.method = 'POST';
        form.action = `/admin/permissions/${id}/clone`;
        
        const csrfToken = document.createElement('input');
        csrfToken.type = 'hidden';
        csrfToken.name = '_token';
        csrfToken.value = document.querySelector('meta[name="csrf-token"]').content;
        
        form.appendChild(csrfToken);
        document.body.appendChild(form);
        form.submit();
    }
}

// Bulk delete permissions
function bulkDeletePermissions() {
    const selected = Array.from(document.querySelectorAll('.permission-checkbox:checked'))
        .map(cb => cb.value);
    
    if (selected.length === 0) {
        alert('Please select at least one permission to delete');
        return;
    }

    if (!confirm(`Delete ${selected.length} selected permission(s)?`)) {
        return;
    }

    fetch('/admin/permissions/bulk-delete', {
        method: 'POST',
        headers: {
            'X-CSRF-TOKEN': document.querySelector('meta[name="csrf-token"]').content,
            'Accept': 'application/json',
            'Content-Type': 'application/json'
        },
        body: JSON.stringify({ ids: selected })
    })
    .then(response => response.json())
    .then(data => {
        if (data.success) {
            location.reload();
        } else {
            alert(data.message);

    /**
     * Display a listing of groups
     */
    public function index(Request $request)
    {
        $query = Group::withCount('permissions');

        // Search functionality
        if ($request->filled('search')) {
            $search = $request->get('search');
            $query->where(function ($q) use ($search) {
                $q->where('name', 'like', "%{$search}%")
                  ->orWhere('slug', 'like', "%{$search}%")
                  ->orWhere('description', 'like', "%{$search}%");
            });
        }

        // Sort functionality
        $sortField = $request->get('sort', 'sort_order');
        $sortDirection = $request->get('direction', 'asc');
        
        if (in_array($sortField, ['name', 'slug', 'sort_order', 'permissions_count', 'created_at'])) {
            $query->orderBy($sortField, $sortDirection);
        }

        $groups = $query->paginate(10)->withQueryString();

        return view('admin.groups.index', compact('groups'));
    }

     * Show the form for creating a new group

    /**
     * Store a newly created group
     */
    public function store(Request $request)
    {
        $validated = $request->validate([
            'name' => 'required|string|max:50',
            'slug' => 'required|string|max:50|unique:permission_groups,slug|regex:/^[a-z0-9\-]+$/',
            'description' => 'nullable|string|max:255',
            'sort_order' => 'nullable|integer|min:0'
        ]);

        // Create group
        $group = Group::create([
            'name' => $validated['name'],
            'slug' => $validated['slug'],
            'description' => $validated['description'],
            'sort_order' => $validated['sort_order'] ?? 0
        ]);

        // Clear cache
        $this->cacheService->forget('groups.all');

        if ($request->input('create_another')) {
            return redirect()
                ->route('admin.groups.create')
                ->with('success', 'Group created successfully! You can add another.');
        }

        return redirect()
            ->route('admin.groups.index')
            ->with('success', 'Group created successfully!');
    }

    /**
     * Update the specified group
     */
    public function update(Request $request, Group $group)
    {
        $validated = $request->validate([
            'name' => 'required|string|max:50',
            'slug' => 'required|string|max:50|unique:permission_groups,slug,' . $group->id . '|regex:/^[a-z0-9\-]+$/',
            'description' => 'nullable|string|max:255',
            'sort_order' => 'nullable|integer|min:0'
        ]);

        // Update group
        $group->update([
            'name' => $validated['name'],
            'slug' => $validated['slug'],
            'description' => $validated['description'],
            'sort_order' => $validated['sort_order'] ?? 0
        ]);

        // Clear cache
        $this->cacheService->forget('groups.all');

        if ($request->input('save_and_continue')) {
            return redirect()
                ->route('admin.groups.edit', $group)
                ->with('success', 'Group updated successfully!');

        return redirect()
            ->route('admin.groups.show', $group)
            ->with('success', 'Group updated successfully!');
    }

    /**
     * Remove the specified group
     */
    public function destroy(Group $group)
    {
        // Check if group has permissions

                'success' => false,
                'message' => 'Cannot delete group with permissions! Move permissions to another group first.'
            ], 403);
        }

        $group->delete();
        
        // Clear cache
        $this->cacheService->forget('groups.all');

        if (request()->wantsJson()) {
            return response()->json([
                'success' => true,
                'message' => 'Group deleted successfully!'
            ]);
        }

        return redirect()
            ->route('admin.groups.index')
            ->with('success', 'Group deleted successfully!');
    }

// Enable drag and drop for reordering
function enableDragAndDrop() {
    const tbody = document.querySelector('#groupsTable tbody');
    if (!tbody) return;
    
    let draggedElement = null;
    
    tbody.querySelectorAll('tr').forEach(row => {
        row.draggable = true;
        row.style.cursor = 'move';
        
        row.addEventListener('dragstart', function(e) {
            draggedElement = this;
            this.style.opacity = '0.5';
        });
        
        row.addEventListener('dragend', function(e) {
            this.style.opacity = '';
        });
        
        row.addEventListener('dragover', function(e) {
            e.preventDefault();
            this.style.backgroundColor = '#f0f0f0';
        });
        
        row.addEventListener('dragleave', function(e) {
            this.style.backgroundColor = '';
        });
        
        row.addEventListener('drop', function(e) {
            e.preventDefault();
            this.style.backgroundColor = '';
            
            if (this !== draggedElement) {
                if (this.parentNode) {
                    this.parentNode.insertBefore(draggedElement, this.nextSibling);
                }
            }
        });
    });
}
